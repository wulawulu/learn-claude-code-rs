impl AgentHook for TodoManager {
    fn on_round_start(&mut self) -> Option<String> {
        self.round_since_update += 1;

            self.round_since_update = 0;
            return Some("<reminder>Refresh your current plan.</reminder>".into());
        }

        None
    }

    fn on_tool_executed(&mut self) {
        self.round_since_update = 0;
    }
}
